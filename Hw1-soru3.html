#include <stdio.h>
#include <math.h>

int main() {
    int popularity;
    int size;
    int value;

    printf("Enter the popularity and size of the house. Popularity and size should be between 1 and 10\\n");

    do {
        printf("popularity:");
        scanf("%d", &popularity);
        printf("size:");
        scanf("%d", &size);

        if (popularity < 1 || popularity > 10 || size < 1 || size > 10) {
            printf("Popularity and size should be scaled between 1 to 10\\n");
        }
    } while (popularity < 1 || popularity > 10 || size < 1 || size > 10);

    value = (pow(popularity, 3) + pow(size, 2)) * 10000;
    printf("House value is= %d", value);

    return 0;
}
